def kmeans_pts (X, Y, M):
P = []
Cls = len(np.unique(Y))
Nc = int(M/Cls)
x_idx = [ [] for _ in range(Cls) ]
for i in range(len(Y)):
x_idx[Y[i]].append(i)
for x in x_idx:
cc = KMeans(n_clusters =
Nc).fit(X[x]).cluster_centers_
px, _ = pairwise_distances_argmin_min(cc,X)
P.append(px)
P = np.concatenate(P).ravel()
return P # prototype result

def kmeans_pts(X, Y, M):
P = []
